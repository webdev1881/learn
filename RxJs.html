RxJs

src="https://cdnjs.cloudflare.com/ajax/libs/rxjs/5.6.0-forward-compat.5/Rx.js"

let stream$ = Rx.Observable.create( function( observer ) { // создаем стрим
 
  console.log('Стрим создан');         // без подписке не создастся
 
  observer.next('One');                // первое событие
 
  observer.complete();                 // форсировано останавливает , без этого третий аргумент подписки не сработает
 
  setTimeout(function() {             // второе событие
    observer.next('After 1 seconds');
  }, 1000)
 
} )
 
stream$
  .subscribe(
   function( data ) { console.log(data)},      // подписка работает вечно (слушает), все события асинхронны
   function( error ) { console.log(error)},    // второй аргумент подписки - ERROR 
   function() { console.log('comleted')}    // третий аргумент подписки - COMLETE 
 )

  var button = document.querySelector('button');
 
Rx.Observable.fromEvent(button, 'click')
  .subscribe(function(e) {
      console.log(e);
  });
 
Rx.Observable.fromEvent(document.querySelector('input'), 'keyup')
   .subscribe(e => console.log(e));
 
Rx.Observable.fromEvent(document, 'mousemove')
  .subscribe(e => {
      document.querySelector('h1').innerHTML = `X: ${e.clientX}, Y: ${e.clientY}`;
  });

Of

Rx.Observable.of(5, 7, '9', 2, [5,4,7,1,2] )  // создаем стрим из любых данныхъ
  .subscribe(
    (x) => console.log(x),
    (err) => console.log(err),
    () => console.log('comlete') // метод OF завершает стрим сам
  )

Interval, Take

Rx.Observable.interval(1000)  // счетчит с нуля с заданым тнтервалом (1000мс)
  .take(5)                     // сколько получить значений из непрерывного потока. На 5 будет complete
  .subscribe(
    (x) => console.log(x),
    (err) => console.log(err),
    () => console.log('comlete') 
  )

Timer, Take

Rx.Observable.timer(500, 1000)  // как Интервал, но с задержкой перед выполнением стрима

  .subscribe(
    (x) => console.log(x),

    () => console.log('comlete')
  )

fromPromise

function delay( ms=1000 ) {
  return new Promise( (res, rej)  => {
    setTimeout(() => {
      res ( ms ) // что бы что то показывать потом
    }, ms)
  }) ;
}

    () => console.log('comlete') 
  )

Map, Pluck

// Map служит для переобразования элементов итерации 
 
 Rx.Observable.fromEvent( document.querySelector('input'), 'keyup' )
  .map( x => {return x.target.value } ) // Метод Map всегда RETURN
  .pluck('target', 'value')             // Метод Pluck - то же самое, только по принципу: ключ-значение
  .map( x => {return x.toUpperCase() } ) // Метод Map можно вызывать сколько угодно
    .subscribe(
      (x) => console.log(x),
      (err) => console.log(err),
      () => console.log('comlete') 
    )
